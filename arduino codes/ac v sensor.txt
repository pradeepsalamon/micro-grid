const int sensorPin = 34;

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
}

void loop() {
  long sum = 0;
  for(int i = 0; i < 1000; i++) {
    int raw = analogRead(sensorPin);
    float v = (raw * 3.3) / 4095.0 - 1.56;
    sum += (v * 1000) * (v * 1000);
    delay(1);
  }
  
  float voltage = (sqrt((float)sum / 1000) / 1000) / 0.001128;
  Serial.println(voltage, 1);
  delay(1000);
}
